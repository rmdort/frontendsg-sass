2 Syntax - Sass & Scss

SASS

.module
  color: #333
  font-weight: 500
  background: #fff
  border: 1px #ccc solid

.sidebar
  @extend .module
  color: #000

SCSS

.module{
  color: #333;
  font-weight: 500;
  background: #fff;
  border: 1px #ccc solid;
}

.sidebar{
  @extend .module;
  color: #000;
}

CSS Output

.module, .sidebar{
  color: #333;
  font-weight: 500;
  background: #fff;
  border: 1px #ccc solid;
}

.sidebar{
  color: #000;
}
  1. Sass syntax uses indentation to indicate nesting of selectors
  2. Scss uses standards CSS syntax with brackets, semicolons

Nesting

Scss

.module{
  a{
    color: white;
    background: red;
    display:inline-block;

    /* Pseudo Hover */

    &:hover{
      color:red;
      background: white;
    }

    /* Target Links with class bullet */

    &.bullet{
      display: list-item;
      list-style: disc inside;
    }    
  }

  /* IE 7 Fix */

  .lt-ie8 &{
    zoom: 1;
  }
}

Css

.module a {
  color: white;
  background: red;
  display: inline-block;
}  

/* Pseudo Hover */  

.module a:hover {
  color: red;
  background: white; 
}

/* Target Links with class bullet */ 

.module a.bullet {
    display: list-item;
    list-style: disc inside; 
}

/* IE 7 Fix */

.lt-ie8 .module {
  zoom: 1;
}

Media Queries

Scss

.module{
  
  width: 500px;
  float: left;

  /* Mobile */

  @media only screen and (min-width: 480px){
    width: auto;
    float: none;
  } 
}

CSS

.module{
  width: 500px;
  float: left;
}

/* Mobile */

@media only screen and (max-width: 480px){
 .module{
   width: auto;
   float: none;
  }
}

Mixins

Scss

/* Border radius Mixin */

.border-radius($radius: 10px){

  border-radius: $radius;
  -moz-border-radius: $radius;
  -webkit-border-radius: $radius;
  -o-border-radius: $radius;
  -ms-border-radius: $radius;

}

/* My Module */

.module{
  @include border-radius(5px);
}

CSS

/* My Module */

.module{
  border-radius: 5px;
  -moz-border-radius: 5px;
  -webkit-border-radius: 5px;
  -o-border-radius: 5px;
  -ms-border-radius: 5px;
}

}

Silent Extend

Scss

/* Error */

/* Presentational markup */

%bordered_box{ 
  border: 1px #ccc solid;
  padding: 10px;
}

/*  - Presentational markup */

.module{
  @extend %bordered_box;
  border-color: red;
}

CSS

/* Error */

.module {
  border: 1px #ccc solid;
  padding: 10px;
}

.module {
  border-color: red;
