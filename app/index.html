Two Syntax - Sass & Scss

SASS (.sass)

.module
  color: #333
  font-weight: 500
  background: #fff
  border: 1px #ccc solid

.sidebar
  @extend .module
  color: #000

SCSS (.scss)

.module{
  color: #333;
  font-weight: 500;
  background: #fff;
  border: 1px #ccc solid;
}

.sidebar{
  @extend .module;
  color: #000;
}

CSS Output (.css)

.module, .sidebar{
  color: #333;
  font-weight: 500;
  background: #fff;
  border: 1px #ccc solid;
}

.sidebar{
  color: #000;
}
  1. Sass syntax uses indentation to indicate nesting of selectors
  2. Scss uses standards CSS syntax with brackets, semicolons

Nesting

Scss

.module{
  a{
    color: white;
    background: red;
    display:inline-block;

    /* Pseudo Hover */

    &:hover{
      color:red;
      background: white;
    }

    /* Target Links with class bullet */

    &.bullet{
      display: list-item;
      list-style: disc inside;
    }    
  }

  /* IE 7 Fix */

  .lt-ie8 &{
    zoom: 1;
  }
}

Css

.module a {
  color: white;
  background: red;
  display: inline-block;
}  

/* Pseudo Hover */  

.module a:hover {
  color: red;
  background: white; 
}

/* Target Links with class bullet */ 

.module a.bullet {
    display: list-item;
    list-style: disc inside; 
}

/* IE 7 Fix */

.lt-ie8 .module {
  zoom: 1;
}

Mixins

Scss

/* REM Font-size */

@mixin border-radius($radius: 10px){

  border-radius: $radius;
  -moz-border-radius: $radius;
  -webkit-border-radius: $radius;
  -o-border-radius: $radius;
  -ms-border-radius: $radius;

}

/* My Module */

.module{
  @include border-radius(0 5px 0 5px);
}

.footer{
  @include border-radius(10px);
}

Functions

Scss

/* Convert Pixels to REM (Given a base font size) */

$base-font-size: 10px;

@function pixel-to-rem($size: $base-font-size){  
  @return $size/ ($base-font-size / 1rem);
}

/* Paragraph */

p{  
  font-size: pixel-to-rem(16px);
}

CSS

/* Paragraph */

p{
  font-size: 1.6rem;
}
